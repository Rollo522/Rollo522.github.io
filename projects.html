def process_lidar(self, msg):
    # Initialise point cloud buffer
    cloud = self.buffer.get_points(msg)
    
    for point in cloud:
        # Apply ground-plane removal filters
        if point.z < -0.5:
            continue
            
        # Cluster centroids for cone detection
        self.map.add_centroid(point.x, point.y)
        
    self.update_perception_state()
